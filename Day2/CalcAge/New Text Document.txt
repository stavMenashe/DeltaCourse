#include <stdio.h>

int main() {
    int curr_day, curr_month, curr_year;
    int birth_day, birth_month, birth_year;
    int age_in_days = 0, age_in_month = 0, age_in_year = 0;

    printf("Please enter the date (format yyyy/mm/dd): ");
    scanf("%d/%d/%d", &curr_year, &curr_month, &curr_day);
    printf("Please enter your birth date (format yyyy/mm/dd): ");
    scanf("%d/%d/%d", &birth_year, &birth_month, &birth_day);

    age_in_days = curr_day - birth_day;
    age_in_month = curr_month - birth_month;
    age_in_year = curr_year - birth_year;

    printf("Your age in days is: %d\n", age_in_days);
    printf("Your age in months is: %d\n", age_in_month);
    printf("Your age in years is: %d\n", age_in_year);
    return 0;
}
